# Problem 72
Collect the nodes of a multiway tree in a list in a bottom-up order. 

```elm
toList tree70 == ['g', 'f', 'c', 'd', 'e', 'b', 'a']
```

![](../i/p70.gif)


```elm

import Html
import List


type MTree a = MNode a (List (MTree a))

          
mtree70 = MNode 'a' 
          [ MNode 'f' [ MNode 'g' [] ]
          , MNode 'c' []
          , MNode 'b' [ MNode 'd' [], MNode 'e' [] ]
          ]

toList : MTree a -> List a
toList (MNode v list) =
    if List.length list == 0 then
        [v]
    else
        (List.concatMap (\x -> toList x) list) ++ [v]


main =
    Html.text
        (if (test) then
            "Your implementation passed all tests."
         else
            "Your implementation failed at least one test."
        )


test : Bool
test =
    List.all ((==) True)
      [ toList (MNode 'a' []) == ['a']
      , toList mtree70 == ['g', 'f', 'c', 'd', 'e', 'b', 'a']
--      , toList test3 == 10

      ]
```
